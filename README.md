# npm_audit
# instalar node js
# Ejecutar la aplicacion con el comando:

    node server.js

# Abrir el navegador y pegar esta dirección    
    
    http://localhost:3000/

# Para ejecutar la auditoria abrir una nueva terminal y ejecutar el comando:    

    npm audit
    npm audit fix
    npm audit fix --force

# si en caso da error al ejecutar ejecutar el comando en power shell como administrador

set-executionpolicy unrestricted
